\begin{vbframe}{Ensembling in Machine Learning}
\begin{itemize}
\item Techniques that combine predictions of multiple base models
\item Goal: Improve predictive performance and model robustness
\item Idea: Errors of a single model will likely be compensated by other models, and as a result, the overall prediction performance of the ensemble would be better than that of a single model
\end{itemize}

\begin{vbframe}{Ensemble Methods}
Higher level mechanisms:
\begin{itemize}
    \item during base model training (how to generate different models?)
    \begin{itemize}
        \item data manipulation (e.g., bagging); partitioning of data
        \item learning algorithm manipulation (e.g., randomly select splits)
        \item output manipulation (error-correcting output codes)
        \item hybridization of the above
    \end{itemize}
    \item during prediction (how to combine the predictions of different models?)
    \begin{itemize}
        \item output manipulation
        \item weighting methods
        \item meta-learning
    \end{itemize}
    \item homogeneous vs. heterogeneous
    \begin{itemize}
        \item same types of models vs. different models
    \end{itemize}
\end{itemize}
\end{vbframe}

\begin{vbframe}{Model Averaging}

Definition: Combine predictions of multiple (different) models to make a final prediction $\rightarrow$ heterogeneous ensembling
\begin{columns}
    \begin{column}{0.5\textwidth}
        \vspace{-1em}
        \begin{figure}
            \centering
            \includegraphics{slides/03-ensembles-stacking/figure/voting.png}
            %\caption{Caption}
            % from https://sebastianraschka.com/pdf/lecture-notes/stat479fs19/07-ensembles__notes.pdf
            \label{fig:voting}
        \end{figure}
    \end{column}
    \begin{column}{0.5\textwidth}
        Steps:
        \begin{enumerate}
            \item Base model training
            \item Combine predictions
        \end{enumerate}
    \end{column}
\end{columns}
\vspace{1em}
Combine predictions: majority voting (hard) / model averaging (soft)
\begin{itemize}
    \item $\fMx = \text{mode}(\{\hblx[1], \ldots, \hblx[M]\})$ assuming hard label predictions    
    \item $\fMx = \frac{1}{M} \sum_{m = 1}^{M} \sblx[m]$ assuming probability predictions
\end{itemize}
\end{vbframe}

% ------------------------------------------------------------------------------

\begin{vbframe}{Weighted Model Averaging}
Assume a set of $M$ different models trained on $\D$.\\
We can also introduce weights $\mathbf{w} = (w_{1}, \ldots, w_{M}), \text{usually: } 0 \le w_{m} \le 1, \sum_{m=1}^{M} w_{m} = 1$ accounting for the contribution of each individual model to the prediction of the ensemble: $\fMwx = \frac{1}{M} \sum_{m = 1}^{M} w_{m} \cdot \sblx[m]$\\
\vspace{1em}
This gives rise to an optimization problem to find the optimal weight vector so a given performance metric $\rho$ of the ensemble is optimal:\\
$\argoptlim_{\mathbf{w} \in \R^{M}} \rho(y, \fMwx)$\\
\vspace{1em}
If we do not know the performance metric or loss function a priori (e.g. AutoML) this optimization problem is treated as a black box
\end{vbframe}

% ------------------------------------------------------------------------------

\begin{vbframe}{Weighted Model Averaging}
Depending on the loss function, the optimization problem can also be straightforward:
\begin{itemize}
    \item Assume MSE as loss function $L(y^{(i)}, \xv^{(i)}) = (y^{(i)} - \fMwxi)^{2}$
    \item And constraints of the form $0 \le w_{m} \le 1, \sum_{m=1}^{M} w_{m} = 1$
    \item Then $\argminlim_{\mathbf{w} \in \R^{M}} \frac{1}{n} \sum_{i = 1}^{n} (y^{(i)} - \fMwxi)^{2}$ s.t. $0 \le w_{m} \le 1, \sum_{m=1}^{M} w_{m} = 1$ can be written as a quadratic program
\end{itemize}
\vspace{1em}
In practice, people typically rely on  
\begin{itemize}
    \item black box optimizers
    \item greedy ensemble selection
    \item linear model $\rightarrow$ stacking
\end{itemize}
\end{vbframe}

\begin{vbframe}{Greedy Ensemble Selection}
Assume a set of $M$ different models trained on $\D$.\\
We now want to optimize the weights used in weighted model averaging.\\
\vspace{1em}
Greedy ensemble selection is one of the most popular methods to do so:
\begin{enumerate}
    \item Start with the empty ensemble
    \item Add the base model that maximizes the ensembles performance metric on a hillclimb (validation) set
    \item Repeat step 2 until a given number of iterations are reached
    \item Return the ensemble from the set of generated ensembles that has the best performance metric on the validation set
\end{enumerate}
\vspace{1em}
Note that ensemble selection effectively optimizes the weights on an integer domain (i.e., base models are selected one or multiple times or not at all)
\end{vbframe}

\begin{vbframe}{Bagging}

Definition: Bagging (stands for Bootstrap Aggregating), training multiple instances of the same base model on different subsets of the training data, obtained through random sampling with replacement $\rightarrow$ homogeneous ensembling
\begin{columns}
    \begin{column}{0.5\textwidth}
        \vspace{-1em}
        \begin{figure}
            \centering
            \includegraphics{slides/03-ensembles-stacking/figure/bagging.png}
            %\caption{Caption}
            % from https://sebastianraschka.com/pdf/lecture-notes/stat479fs19/07-ensembles__notes.pdf
            \label{fig:bagging}
        \end{figure}
    \end{column}
    \begin{column}{0.5\textwidth}
        Steps:
        \begin{enumerate}
            \item Random sampling with replacement to generate different train data sets
            \item Base model training
            \item Combine predictions
        \end{enumerate}
    \end{column}
\end{columns}
\vspace{1em}
Popular use case: Construction of a random forest using decision trees as base models

\end{vbframe}

% ------------------------------------------------------------------------------

\begin{vbframe}{Bagging}

Why subsets of training data obtained through random sampling with replacement (input manipulation):
\begin{itemize}
    \item Bias vs. variance trade-off
    \begin{itemize}
        \item By training each model on slightly different data the variance of the predictions are reduced
        \item The training data is reused many times so bias can be slightly higher  
    \end{itemize}
    \item Effective for cases where small changes in the training set may result in a completely different model (instable learners)
\end{itemize}
\vspace{1em}
Note that a random forest does not only employ input manipulation via bagging but also learning algorithm manipulation (sampling splits).

\end{vbframe}

\begin{vbframe}{Boosting}
Definition: Boosting is an iterative technique that adjusts the weights of the training instances based on their misclassification rate or residual errors, and sequentially trains subsequent base models (weak learners) to focus on the misclassified or poorly predicted instances.
\vspace{1em}
Well known frameworks:
\begin{itemize}
    \item AdaBoost (Adaptive Boosting)
    \item Gradient Boosting
    \item XGBoost (Extreme Gradient Boosting)
\end{itemize}
\vspace{1em}
Different from other ensembling methods as training is done sequentially

\begin{vbframe}{Stacking}
Definition: Stacking (also stacked generalization or meta-ensemble), is an ensemble technique that involves training multiple base models (level 0) on the same data and using their predictions as input features for a level 1 model that makes the final ensemble prediction $\rightarrow$ heterogeneous ensembling

\begin{columns}
    \begin{column}{0.5\textwidth}
        \vspace{-1em}
        \begin{figure}
            \centering
            \includegraphics[width=0.9\textwidth]{slides/03-ensembles-stacking/figure/stacking.png}
            %\caption{Caption}
            % from https://sebastianraschka.com/pdf/lecture-notes/stat479fs19/07-ensembles__notes.pdf
            \label{fig:stacking}
        \end{figure}
    \end{column}
    \begin{column}{0.5\textwidth}
        Steps:
        \begin{enumerate}
            \item Base model training (level 0)
            \item Collect predictions
            \item level 1 model training
        \end{enumerate}
    \end{column}
\end{columns}
\vspace{1em}

In contrast to other methods, stacking already combines predictions during training.

\end{vbframe}

% ------------------------------------------------------------------------------

\begin{vbframe}{Stacking}
\footnotesize
\begin{algorithm}[H]
    \SetAlgoLined
    \caption{Stacking}
    \SetKwInOut{Input}{Input}
    \SetKwInOut{Output}{Output}

    \Input{$\D = \Dset$}
    \Output{Ensemble}
    
    \BlankLine
    \textbf{Step 1: Learn level 0 base models}\;\\
    \For{$m \leftarrow 1$ \KwTo $M$}{
        Fit base model $\bl[m]$ on $\D$\;
    }
    
    \BlankLine
    \textbf{Step 2: Construct new dataset $\D^\prime$ from $\D$}\;\\
    \For{$i \leftarrow 1$ \KwTo $n$}{
        Add $(\mathbf{x}^{\prime(i)}, y^{(i)})$ to new dataset, where $\mathbf{x}^{\prime(i)} = (\blx[1], \ldots, \blx[M])$\;
    }
    
    \BlankLine
    \textbf{Step 3: Learn level 1 model $\fM(\D^\prime)$}\;\\
    \KwRet $\fM(\D^\prime)$\;
\end{algorithm}
Note: We can either drop the original features or keep them when learning the level 1 model

\end{vbframe}

\begin{vbframe}{Multilevel Stacking}
So far: train level 0 base models, use predictions to train the level 1 model.\\
Now: train multiple models at level 1, use predictions to train final level 2 model 

\begin{columns}
    \begin{column}{0.5\textwidth}

        \begin{enumerate}
            \item level 0 base model training
            \item Collect predictions of level 0 base models
            \item level 1 model training 
            \item Collect predictions of level 1 models
            \item level 2 final model training
        \end{enumerate}
    \end{column}
\end{columns}
\vspace{1em}
\end{vbframe}
