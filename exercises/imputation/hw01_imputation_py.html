from sklearn.experimental import enable_iterative_imputer
from sklearn.ensemble import RandomForestRegressor
from sklearn.impute import IterativeImputer, KNNImputer
from sklearn.linear_model import LinearRegression
from sklearn.model_selection import GridSearchCV
from sklearn.pipeline import Pipeline
from sklearn.base import BaseEstimator, TransformerMixin

class BranchImputer(BaseEstimator, TransformerMixin):
    # wrapper that branches between different imputation methods
    def __init__(self, method="linear", n_neighbors=5, random_state=None):
        self.method = method               # "linear" or "knn"
        self.n_neighbors = n_neighbors     # only for knn
        self.random_state = random_state

    def fit(self, X, y=None):
        # choose imputer based on method
        if self.method == "linear":
            self.impl_ = IterativeImputer(
                estimator=LinearRegression(),
                initial_strategy='most_frequent',
                random_state=self.random_state,
                imputation_order='random'
            )
        elif self.method == "knn":
            self.impl_ = KNNImputer(
                n_neighbors=self.n_neighbors
            )
        else:
            raise ValueError(f"Unknown method: {self.method}")

        self.impl_.fit(X)
        return self

    def transform(self, X):
        return self.impl_.transform(X)

pipeline = Pipeline([
    ('imputer', BranchImputer(method="linear", random_state=12345)),
    ('model', RandomForestRegressor(random_state=12345))
])

# Extract and display the results from GridSearchCV
import pandas as pd
import matplotlib.pyplot as plt

# Convert the CV results to a DataFrame for easier analysis
results = pd.DataFrame(search.cv_results_)

# Extract the MSE values (convert from negative to positive)
results['mean_test_mse'] = -results['mean_test_score']
results['std_test_mse'] = results['std_test_score']

# Print the MSE for each method
print("Mean Squared Error (MSE) for each imputation method:")
print("-" * 50)

# Linear imputation result
linear_result = results[results['param_imputer__method'] == 'linear']
print(f"Linear imputation MSE: {linear_result['mean_test_mse'].values[0]:.4f} ± {linear_result['std_test_mse'].values[0]:.4f}")

# KNN imputation results
knn_results = results[results['param_imputer__method'] == 'knn']
print("\nKNN imputation MSE by n_neighbors:")
for _, row in knn_results.iterrows():
    n_neighbors = row['param_imputer__n_neighbors']
    mse = row['mean_test_mse']
    std = row['std_test_mse']
    print(f"n_neighbors = {n_neighbors}: {mse:.4f} ± {std:.4f}")

plt.figure(figsize=(15, 5))

# Plot for KNN with different n_neighbors
plt.subplot(1, 2, 1)
n_neighbors = knn_results['param_imputer__n_neighbors'].values
mse_values = knn_results['mean_test_mse'].values
std_values = knn_results['std_test_mse'].values

plt.errorbar(n_neighbors, mse_values, yerr=std_values, marker='o', linestyle='-')
plt.axhline(y=linear_result['mean_test_mse'].values[0], color='r', linestyle='--', label='Linear imputation')
plt.xlabel('Number of neighbors (KNN)')
plt.ylabel('Mean Squared Error')
plt.title('KNN Imputation Performance by n_neighbors')
plt.grid(True)
plt.legend()

plt.subplot(1, 2, 2)
